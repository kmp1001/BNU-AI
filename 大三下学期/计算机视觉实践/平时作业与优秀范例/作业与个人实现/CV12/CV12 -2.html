# draw hand landmarks on a image
def draw_landmarks_on_image(img, detection_result):
    h,w,c = img.shape

    for one_hand in detection_result.hand_landmarks:
        for id, lm in enumerate(one_hand):
            x, y = int(w*lm.x), int(h*lm.y) #lm.x/y
            print(id, x, y) #
            if id==4:
                cv2.circle(img, (x, y), 10, (0,255,0), 2)

        hand_landmarks_proto = landmark_pb2.NormalizedLandmarkList()
        hand_landmarks_proto.landmark.extend([
        landmark_pb2.NormalizedLandmark(x=lm.x, y=lm.y, z=lm.z) for lm in one_hand])
        solutions.drawing_utils.draw_landmarks(img, hand_landmarks_proto, mp.solutions.hands.HAND_CONNECTIONS)

            depth_image = np.asanyarray(depth_frame.get_data()) #Convert to numpy array

    objDict = {}

def draw_landmarks_on_image(rgb_image, detection_result): #hand landmarks
    hand_landmarks_list = detection_result.hand_landmarks
    handedness_list = detection_result.handedness
    annotated_image = np.copy(rgb_image)

    # Loop through the detected hands to visualize.
    for idx in range(len(hand_landmarks_list)):  #
        hand_landmarks = hand_landmarks_list[idx]   #xy
        handedness = handedness_list[idx]  #

        # Draw the hand landmarks.
        hand_landmarks_proto = landmark_pb2.NormalizedLandmarkList()
        hand_landmarks_proto.landmark.extend([
          landmark_pb2.NormalizedLandmark(x=landmark.x, y=landmark.y, z=landmark.z) for landmark in hand_landmarks
        ])

          hand_landmarks_proto,
          solutions.hands.HAND_CONNECTIONS,
          solutions.drawing_styles.get_default_hand_landmarks_style(),
          solutions.drawing_styles.get_default_hand_connections_style())

    return annotated_image


def get_finger_xyz(detection_result, depth_img, intrinsic): #hand landmark
    finger_xyz = {}
    if len(detection_result.hand_landmarks) > 0:
        hand_landmarks_list = detection_result.hand_landmarks
        handedness_list = detection_result.handedness

        for idx in range(len(hand_landmarks_list)):  #

            handedness = handedness_list[idx]  # 

            lm_xys = [[int(wCam * hand_landmarks[i].x), int(hCam * hand_landmarks[i].y)] for i in
                      finger_lm_idx]

            xyzs = []
            center_x, center_y, center_z = 0,0,0
            for x, y in lm_xys:
                if (0 <= x < wCam) and (0 <= y < hCam):
                    depth = depth_img[y, x]
                    if depth > 0:
                        xyz = rs.rs2_deproject_pixel_to_point(intrinsic, [x, y], depth)
                        xyzs.append(xyz)
                        center_x += xyz[0]
                        center_y += xyz[1]
                        center_z += xyz[2]
            if len(xyzs):
                center_x /= len(xyzs)
                center_y /= len(xyzs)
                center_z /= len(xyzs)
            xyzs.append([center_x, center_y, center_z])  #

            if handedness[0].category_name == 'Left':
                finger_xyz['left'] = xyzs
            else:
                finger_xyz['right'] = xyzs
    return finger_xyz
